22 // CPU-side Novosibirsk evaluation for use in generating toy MC.
23 double novosib(double x, double peak, double width, double tail) {
24  double qa = 0, qb = 0, qc = 0, qx = 0, qy = 0;
25 
26  if(fabs(tail) < 1.e-7)
27  qc = 0.5 * pow(((x - peak) / width), 2);
28  else {
29  qa = tail * sqrt(log(4.));
30  qb = sinh(qa) / qa;
31  qx = (x - peak) / width * qb;
32  qy = 1. + tail * qx;
33 
34  //---- Cutting curve from right side
35  if(qy > 1.E-7)
36  qc = 0.5 * (pow((log(qy) / tail), 2) + tail * tail);
37  else
38  qc = 15.0;
39  }
40 
41  //---- Normalize the result
42  return exp(-qc);
43 }
